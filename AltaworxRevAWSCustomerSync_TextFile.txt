CREATE PROCEDURE [dbo].[usp_Update_RevCustomer]  
    @IntegrationAuthenticationId INT  
AS  
    DECLARE @processedBy NVARCHAR(50) = 'usp_Update_RevCustomer';  
  
    -- IF Staging table is empty, then exit  
    IF NOT EXISTS (SELECT 1 FROM [dbo].[RevCustomerStaging] WHERE [IntegrationAuthenticationId] = @IntegrationAuthenticationId)  
    BEGIN  
        RETURN;  
    END  
  
    MERGE [dbo].[RevCustomer] AS [TARGET]  
    USING (  
        SELECT [RevCustomerId],  
            [CustomerName],  
            [ParentCustomerId],  
            [IntegrationAuthenticationId],  
            [Status],  
            [ActivatedDate],  
            [CloseDate],  
            [TaxExemptEnabled],  
            [TaxExemptTypes],  
            [BillProfileId],  
            [AgentId]  
        FROM [dbo].[RevCustomerStaging]  
    ) AS [SOURCE]  
    ON [TARGET].[RevCustomerId] = [SOURCE].[RevCustomerId] AND [TARGET].[IntegrationAuthenticationId] = [SOURCE].[IntegrationAuthenticationId]  
    WHEN MATCHED THEN  
        UPDATE  
        SET [TARGET].[CustomerName] = [SOURCE].[CustomerName],  
            [TARGET].[RevParentCustomerId] = [SOURCE].[ParentCustomerId],  
            [TARGET].[IntegrationAuthenticationId] = [SOURCE].[IntegrationAuthenticationId],  
            [TARGET].[Status] = [SOURCE].[Status],  
            [TARGET].[ActivatedDate] = [SOURCE].[ActivatedDate],  
            [TARGET].[CloseDate] = [SOURCE].[CloseDate],  
            [TARGET].[ModifiedBy] = @processedBy,  
            [TARGET].[ModifiedDate] = GETUTCDATE(),  
            [TARGET].[TaxExemptEnabled] = [SOURCE].[TaxExemptEnabled],  
            [TARGET].[TaxExemptTypes] = [SOURCE].[TaxExemptTypes],  
            [TARGET].[IsActive] = CASE WHEN [SOURCE].[Status] = 'CLOSED' THEN 0 ELSE 1 END,  
            [TARGET].[IsDeleted] = CASE WHEN [SOURCE].[Status] = 'CLOSED' THEN 1 ELSE 0 END,  
            [TARGET].[BillProfileId] = [SOURCE].[BillProfileId],  
            [TARGET].[AgentId] = [SOURCE].[AgentId]  
    WHEN NOT MATCHED BY TARGET THEN  
        INSERT (  
            [RevCustomerId],  
            [CustomerName],  
            [RevParentCustomerId],  
            [CreatedBy],[CreatedDate],  
            [IsDeleted],  
            [IntegrationAuthenticationId],  
            [Status],  
            [ActivatedDate],  
            [CloseDate],  
            [TaxExemptEnabled],  
            [TaxExemptTypes],  
            [BillProfileId],  
            [AgentId]  
        )  
        VALUES (  
            [SOURCE].[RevCustomerId],  
            [SOURCE].[CustomerName],  
            [SOURCE].[ParentCustomerId],  
            @processedBy,  
            GETUTCDATE(),  
            0,  
            [SOURCE].[IntegrationAuthenticationId],  
            [SOURCE].[Status],  
            [SOURCE].[ActivatedDate],  
            [SOURCE].[CloseDate],  
            [SOURCE].[TaxExemptEnabled],  
            [SOURCE].[TaxExemptTypes],  
            [SOURCE].[BillProfileId],  
            [SOURCE].[AgentId]  
        )  
    WHEN NOT MATCHED BY SOURCE AND [TARGET].[IntegrationAuthenticationId] = @IntegrationAuthenticationId THEN  
        UPDATE  
     SET  
         [TARGET].[IsDeleted] = 1,  
         [TARGET].[IsActive] = 0,  
         [TARGET].[ModifiedBy] = @processedBy,  
         [TARGET].[ModifiedDate] = GETUTCDATE();  
  
    UPDATE [RevCustomer]  
    SET [ParentCustomerId] = [ParentRevCustomer].[id]  
    FROM [RevCustomer]  
    LEFT JOIN [RevCustomer] [ParentRevCustomer] ON [RevCustomer].[RevParentCustomerId] = [ParentRevCustomer].[RevCustomerId] AND [RevCustomer].[IntegrationAuthenticationId] = [ParentRevCustomer].[IntegrationAuthenticationId];  
  
    UPDATE [RevCustomer]  
    SET [ChildCount] = ((SELECT COUNT(1) FROM [RevCustomer] WHERE [RevCustomer].[RevCustomerId] = [RevParentCustomerId] AND [RevCustomer].[IntegrationAuthenticationId] = [IntegrationAuthenticationId] AND [Status] <> 'CLOSED'))  
    FROM [RevCustomer];  
  
    --=================================================  
    -- Case Site record does not exist  
    -- Unique constraint [UC_Site] on   
    -- [TenantId]  
    -- [Name]  
    --  
    -- NOTE: Because we have many RevCustomer records that have the same customer names with different RevCustomerIDs,  
    --       we are appending RevCustomerIDs to CustomerName.  
    --  
    -- E.g. Bluefox.io has 3 entries with 3 different RevCustomerIds  
    --  
    -- RevCustomerId CustomerName  
    -- 300007255 Bluefox.io  
    -- 300006635 Bluefox.io  
    -- 300006784 Bluefox.io  
    --  
    -- [dbo].[Site] will have:  
    --   
    -- RevCustomerId Name   
    -- 300007255 Bluefox.io - 300007255  
    -- 300006635 Bluefox.io - 300006635  
    -- 300006784 Bluefox.io - 300006784  
  
    --=================================================  
    INSERT INTO [dbo].[Site] (  
            [TenantId],  
            [Name],  
            [RevCustomerId],  
            [CreatedDate],  
            [CreatedBy],  
            [IsDeleted],  
            [IsActive]  
        )  
    SELECT DISTINCT  
        [IntegrationAuthentication].[TenantId],  
        [RevCustomer].[CustomerName] + ' (' + [RevCustomer].[RevCustomerId] + ')' AS [Name],  
        [RevCustomer].[Id] AS [RevCustomerId],  
        GETUTCDATE() AS [CreatedDate],  
        'RevSync' AS [CreatedBy],  
        0,  
        1  
    FROM [dbo].[RevCustomer]  
    INNER JOIN [dbo].[Integration_Authentication] [IntegrationAuthentication] ON [RevCustomer].[IntegrationAuthenticationId] = [IntegrationAuthentication].[Id]  
    WHERE NOT EXISTS (  
            SELECT 1   
            FROM [dbo].[Site]   
            WHERE   
                Site.[RevCustomerId] = [RevCustomer].[Id]  
                AND  
                Site.[TenantId] = [IntegrationAuthentication].[TenantId]  
        );  
  
    --========================================================================  
    -- Update default customer billing period end day and end hour  
    --========================================================================  
    UPDATE [Site]  
    SET  
        [CustomerBillPeriodEndDay] = IIF(([Site].[CustomerBillPeriodEndDay] IS NULL OR ([Site].[CustomerBillPeriodEndDay] IS NOT NULL AND [Site].[CustomerBillPeriodEndDay] != [RevCustomerStaging].[CustomerBillPeriodEndDay])) AND [RevCustomerStaging].[Cust
omerBillPeriodEndDay] <= 28 AND [RevCustomerStaging].[CustomerBillPeriodEndDay] >= 1, [RevCustomerStaging].[CustomerBillPeriodEndDay], [Site].[CustomerBillPeriodEndDay]),  
        [CustomerBillPeriodEndHour] = IIF(([Site].[CustomerBillPeriodEndHour] IS NULL OR ([Site].[CustomerBillPeriodEndHour] IS NOT NULL AND [Site].[CustomerBillPeriodEndHour] != [RevCustomerStaging].[CustomerBillPeriodEndHour])) AND [RevCustomerStaging].
[CustomerBillPeriodEndHour] <= 23 AND [RevCustomerStaging].[CustomerBillPeriodEndHour] >= 0, [RevCustomerStaging].[CustomerBillPeriodEndHour], [Site].[CustomerBillPeriodEndHour])  
    FROM  
        [dbo].[Site]  
  INNER JOIN [dbo].[RevCustomer] ON [RevCustomer].[id] = [Site].[RevCustomerId] AND [RevCustomer].[IntegrationAuthenticationId] = @IntegrationAuthenticationId  
        INNER JOIN [dbo].[RevCustomerStaging] ON [RevCustomerStaging].[RevCustomerId] = [RevCustomer].[RevCustomerId] AND [RevCustomerStaging].[IntegrationAuthenticationId] = [RevCustomer].[IntegrationAuthenticationId]  
    WHERE   
       [Site].[CustomerBillPeriodEndDay] IS NULL   
  OR ([Site].[CustomerBillPeriodEndDay] IS NOT NULL AND [Site].[CustomerBillPeriodEndDay] != [RevCustomerStaging].[CustomerBillPeriodEndDay])  
  OR [Site].[CustomerBillPeriodEndHour] IS NULL  
  OR ([Site].[CustomerBillPeriodEndHour] IS NOT NULL AND [Site].[CustomerBillPeriodEndHour] != [RevCustomerStaging].[CustomerBillPeriodEndHour])  
  
    --========================================================================  
    -- Update Flag Delete Of Site When RevCustomer Is Closed  
    --========================================================================  
    UPDATE [Site]  
    SET  
        [IsActive] = [RevCustomer].[IsActive],  
        [IsDeleted] = [RevCustomer].[IsDeleted],  
        [ModifiedBy] = 'RevSync',  
        /* Keeping it consitent with Jasper Sync*/  
        [ModifiedDate] = GETDATE()  
    FROM  
        [dbo].[Site]  
        INNER JOIN [dbo].[RevCustomer] ON [RevCustomer].[id] = [Site].[RevCustomerId] AND [RevCustomer].[IntegrationAuthenticationId] = @IntegrationAuthenticationId  
        INNER JOIN [dbo].[Integration_Authentication] ON [RevCustomer].[IntegrationAuthenticationId] = [Integration_Authentication].[id] AND [Site].[TenantId] = [Integration_Authentication].[TenantId];  
    --========================================================================  
    -- Update Device_Tenant and Mobility_DeviceTenant to Default site(Unassigned) When RevCustomer Is Closed  
    --========================================================================  
 UPDATE [Device_Tenant]  
 SET  
  [SiteId] = CASE WHEN [RevCustomer].[Status] = 'CLOSED' THEN [Site].[id] ELSE [Device_Tenant].[SiteId] END,  
  [ModifiedBy] = 'RevSync',  
        /* Keeping it consitent with Jasper Sync*/  
        [ModifiedDate] = GETDATE()  
 FROM   
  [dbo].[Device_Tenant]  
  INNER JOIN [dbo].[RevCustomer] ON [RevCustomer].[RevCustomerId] = [Device_Tenant].AccountNumber AND [RevCustomer].IntegrationAuthenticationId = @IntegrationAuthenticationId  
  INNER JOIN [dbo].[Site] ON [IsSystemDefault] = 1 AND [Device_Tenant].[TenantId] = [Site].[TenantId] AND [Site].IsActive = 1 AND [Site].[IsDeleted] = 0  
 WHERE [Device_Tenant].[IsDeleted] = 0 AND [Device_Tenant].[IsActive] = 1  
   AND [Device_Tenant].AccountNumberIntegrationAuthenticationId = @IntegrationAuthenticationId  
 UPDATE [MobilityDevice_Tenant]  
 SET  
  [SiteId] = CASE WHEN [RevCustomer].[Status] = 'CLOSED' THEN [Site].[id] ELSE [MobilityDevice_Tenant].[SiteId] END,  
  [ModifiedBy] = 'RevSync',  
        /* Keeping it consitent with Jasper Sync*/  
        [ModifiedDate] = GETDATE()  
 FROM   
  [dbo].[MobilityDevice_Tenant]  
  INNER JOIN [dbo].[RevCustomer] ON [RevCustomer].[RevCustomerId] = [MobilityDevice_Tenant].AccountNumber AND [RevCustomer].IntegrationAuthenticationId = @IntegrationAuthenticationId  
  INNER JOIN [dbo].[Site] ON [IsSystemDefault] = 1 AND [MobilityDevice_Tenant].[TenantId] = [Site].[TenantId] AND [Site].IsActive = 1 AND [Site].[IsDeleted] = 0  
    WHERE [MobilityDevice_Tenant].[IsDeleted] = 0 AND [MobilityDevice_Tenant].[IsActive] = 1  
AND [MobilityDevice_Tenant].AccountNumberIntegrationAuthenticationId = @IntegrationAuthenticationId  
    --========================================================================  
    -- Case Site record exists and Name is different  
    --========================================================================  
    UPDATE [Site]  
    SET  
        [Name] = [RevCustomer].[CustomerName] + ' (' + [RevCustomer].[RevCustomerId] + ')',  
        [ModifiedBy] = 'RevSync',  
        /* Keeping it consitent with Jasper Sync*/   
        [ModifiedDate] = GETDATE()   
    FROM  
        [dbo].[Site]  
        INNER JOIN [dbo].[RevCustomer] ON [RevCustomer].[id] = [Site].[RevCustomerId]  
    WHERE  
        /* null is not allowed both for Site.Name and [RevCustomerStaging].RevCustomerId so this check is sufficient. */  
        [Site].[Name] != [RevCustomer].[CustomerName] + ' (' + [RevCustomer].[RevCustomerId] + ')';  
  
    --========================================================================  
    -- Case new Child accounts created to be automatically assigned to the Customer Group  
    --========================================================================  
    INSERT INTO [dbo].[SiteGroup_Site] ([SiteGroupId], [SiteId])  
    SELECT [SiteGroupSite].[SiteGroupId], [Site].[id]  
    FROM  
        [dbo].[Site]  
        INNER JOIN [dbo].[RevCustomer] ON [RevCustomer].[id] = [Site].[RevCustomerId] AND [RevCustomer].[IntegrationAuthenticationId] = @IntegrationAuthenticationId

    SET  
        [ParentSiteId] = [ParentSite].[id],  
        [ModifiedBy] = 'RevSync',  
        [ModifiedDate] = GETDATE()  
    FROM  
        [dbo].[Site]  
        INNER JOIN [dbo].[RevCustomer] ON [RevCustomer].[id] = [Site].[RevCustomerId] AND [RevCustomer].[IntegrationAuthenticationId] = @IntegrationAuthenticationId  
        LEFT JOIN [dbo].[Site] [ParentSite] ON [ParentSite].[RevCustomerId] = [RevCustomer].[ParentCustomerId]

CREATE PROCEDURE [dbo].[usp_Update_RevProductType_FromStaging]  
    @IntegrationAuthenticationId INT   
AS  
BEGIN  
    SET NOCOUNT ON;  
  
    -- IF Staging table is empty, then exit  
    IF NOT EXISTS (SELECT 1 FROM [RevProductTypeStaging] WHERE IntegrationAuthenticationId = @IntegrationAuthenticationId)  
    BEGIN  
        RETURN;  
    END  
  
    DECLARE @processedBy NVARCHAR(50) = 'usp_Update_RevProductType_FromStaging';  
    MERGE [dbo].[RevProductType] AS [TARGET]  
    USING (  
        SELECT [ProductTypeId],  
            [ProductTypeCode],  
            [Description],  
            [TaxClassId],  
            [IsActive],  
            [IntegrationAuthenticationId]  
        FROM  
        (  
            SELECT [RevProductTypeStaging].[ProductTypeId],  
                [RevProductTypeStaging].[ProductTypeCode],  
                [RevProductTypeStaging].[Description],  
                [RevProductTypeStaging].[TaxClassId],  
                [RevProductTypeStaging].[IsActive],  
                [RevProductTypeStaging].[IntegrationAuthenticationId],  
                ROW_NUMBER() OVER(PARTITION BY [ProductTypeId] ORDER BY [RevProductTypeStaging].[RevProductTypeStagingId] DESC) AS [RecordNumber]  
            FROM [dbo].[RevProductTypeStaging]  
        ) [RevProductTypeStg]  
        WHERE [RecordNumber] = 1  
    ) AS [SOURCE]  
    ON [TARGET].[ProductTypeId] = [SOURCE].[ProductTypeId] AND [TARGET].[IntegrationAuthenticationId] = [SOURCE].[IntegrationAuthenticationId]  
    WHEN NOT MATCHED BY TARGET THEN  
        INSERT (  
      [ProductTypeId],  
      [ProductTypeCode],  
      [Description],  
      [TaxClassId],  
      [CreatedBy],  
      [CreatedDate],  
      [ModifiedBy],  
      [ModifiedDate],  
      [DeletedBy],  
      [DeletedDate],  
      [IsDeleted],  
      [IsActive],  
      [IntegrationAuthenticationId]  
     )  
     VALUES (  
      [SOURCE].[ProductTypeId],  
      [SOURCE].[ProductTypeCode],  
      [SOURCE].[Description],  
      [SOURCE].[TaxClassId],  
      @processedBy,  
      GETUTCDATE(),  
      NULL,  
      NULL,  
      NULL,  
      NULL,  
      0,  
      [SOURCE].[IsActive],  
      [SOURCE].[IntegrationAuthenticationId]  
        )  
    WHEN NOT MATCHED BY SOURCE AND [TARGET].[IntegrationAuthenticationId] = @IntegrationAuthenticationId THEN  
        UPDATE  
     SET  
         [TARGET].[IsDeleted] = 1,  
         [TARGET].[IsActive] = 0,  
         [TARGET].[ModifiedBy] = @processedBy,  
         [TARGET].[ModifiedDate] = GETUTCDATE()  
    WHEN MATCHED THEN  
        UPDATE  
        SET  
            [TARGET].[ProductTypeCode] = [SOURCE].[ProductTypeCode],

                [RevServiceTypeStaging].[ServiceTypeId]  
    ,[RevServiceTypeStaging].[Description]  
    ,[RevServiceTypeStaging].[IsActive]

    IF NOT EXISTS (SELECT 1 FROM [RevServiceStaging] WHERE IntegrationAuthenticationId = @IntegrationAuthenticationId)  
    BEGIN  
        RETURN;  
    END  
  
    DECLARE @processedBy NVARCHAR(50) = 'usp_Update_RevService';  
    MERGE [dbo].[RevService] AS [TARGET]  
    USING  
    (  
        SELECT [RevCustomerId],  
            [Number],  
            [RevServiceId],  
            [RatePlanCode],  
            [ActivatedDate],  
            [DisconnectedDate],  
            [IntegrationAuthenticationId],  
            [RevProviderId],  
            [RevServiceTypeId],

                [Number],  
                [RevServiceId],  
                [RevServiceStaging].[RatePlanCode],  
                [RevServiceStaging].[ActivatedDate],  
                [RevServiceStaging].[DisconnectedDate],

                ROW_NUMBER() OVER(PARTITION BY [RevServiceId], [RevServiceStaging].[IntegrationAuthenticationId] ORDER BY [Number]) AS [RowNumber],

            INNER JOIN [RevCustomer] ON [RevServiceStaging].[RevCustomerId] = [RevCustomer].[RevCustomerId] AND [RevServiceStaging].[IntegrationAuthenticationId] = [RevCustomer].[IntegrationAuthenticationId]  
            LEFT JOIN [dbo].[RevProvider] ON [RevServiceStaging].[RevProviderId] = [RevProvider].[ProviderId] AND [RevServiceStaging].[IntegrationAuthenticationId] = [RevProvider].[IntegrationAuthenticationId]  
            LEFT JOIN [dbo].[RevServiceType] ON [RevServiceStaging].[ServiceTypeId] = [RevServiceType].[ServiceTypeId] AND [RevServiceStaging].[IntegrationAuthenticationId] = [RevServiceType].[IntegrationAuthenticationId]  
        ) [RevServiceStg]  
        WHERE [RowNumber] = 1  
    ) AS [SOURCE]  
    ON [TARGET].RevServiceId = [SOURCE].RevServiceId AND [TARGET].[IntegrationAuthenticationId] = [SOURCE].[IntegrationAuthenticationId]  
    WHEN NOT MATCHED BY TARGET THEN  
        INSERT(  
            [RevCustomerId]  
            ,[Number]  
            ,[RevServiceId]  
            ,[RatePlanCode]  
            ,[ActivatedDate]  
            ,[DisconnectedDate]  
            ,[CreatedBy]  
            ,[CreatedDate]  
            ,[IsDeleted]  
            ,[IsActive]  
            ,[IntegrationAuthenticationId]  
            ,[RevProviderId]  
            ,[RevServiceTypeId]  
            ,[Description]  
        )  
        VALUES(  
            [SOURCE].[RevCustomerId]  
            ,[SOURCE].[Number]  
            ,[SOURCE].[RevServiceId]  
            ,[SOURCE].[RatePlanCode]  
            ,[SOURCE].[ActivatedDate]  
            ,[SOURCE].[DisconnectedDate]  
            ,@processedBy  
            ,GETUTCDATE()  
            ,0  
            ,1  
            ,[SOURCE].[IntegrationAuthenticationId]  
            ,[SOURCE].[RevProviderId]  
            ,[SOURCE].[RevServiceTypeId]  
            ,[SOURCE].[Description]  
        )  
    WHEN NOT MATCHED BY SOURCE AND [TARGET].[IntegrationAuthenticationId] = @IntegrationAuthenticationId THEN  
        UPDATE  
        SET  
            [TARGET].[IsDeleted] = 1,  
            [TARGET].[IsActive] = 0,  
            [TARGET].[ModifiedBy] = @processedBy,  
            [TARGET].[ModifiedDate] = GETUTCDATE()

                        TARGET.[Number] <> SOURCE.[Number]  
                        OR TARGET.[RevCustomerId] <> SOURCE.[RevCustomerId]   
                        OR TARGET.[RatePlanCode] <> SOURCE.[RatePlanCode]   
                        OR TARGET.[ActivatedDate] <> SOURCE.[ActivatedDate]  
                     OR TARGET.[DisconnectedDate] <> SOURCE.[DisconnectedDate]  
                        OR TARGET.[IntegrationAuthenticationId] <> SOURCE.[IntegrationAuthenticationId]  
      OR TARGET.[RevProviderId] <> SOURCE.[RevProviderId]  
      OR TARGET.[RevServiceTypeId] <> SOURCE.[RevServiceTypeId]  
      OR TARGET.[Description] <> SOURCE.[Description]  
      OR (TARGET.IsDeleted = 1)  
      OR (TARGET.IsActive = 0)) THEN

            [TARGET].[RevCustomerId] = [SOURCE].[RevCustomerId],  
            [TARGET].[RatePlanCode] = [SOURCE].[RatePlanCode],  
            [TARGET].[ActivatedDate] = [SOURCE].[ActivatedDate],  
            [TARGET].[DisconnectedDate] = [SOURCE].[DisconnectedDate],  
            [TARGET].[ModifiedBy] = @processedBy,  
            [TARGET].[ModifiedDate] = GETUTCDATE(),

            [TARGET].[RevProviderId] = [SOURCE].[RevProviderId],  
            [TARGET].[RevServiceTypeId] = [SOURCE].[RevServiceTypeId],

            [RevServiceId],  
            [DisconnectedDate]  
        FROM  
        (  
            SELECT   
                [MobilityDeviceId],  
                [TenantId],  
                [SiteId],  
                [AccountNumber],  
                [IntegrationAuthenticationId],  
                [RevServiceId],  
                [Number],  
                [ActivatedDate],  
                [DisconnectedDate],  
                ROW_NUMBER() OVER(PARTITION BY [IntegrationAuthenticationId], [MobilityDeviceId] ORDER BY [Status], [DisconnectedDate] DESC, [ActivatedDate] DESC) AS [RecordNumber]  
            FROM  
                (  
                SELECT [MobilityDevice].[id] AS [MobilityDeviceId],  
                    [Site].[TenantId],  
                    [Site].[id] AS [SiteId],  
                    [RevCustomer].[RevCustomerId] AS [AccountNumber],  
                    [RevCustomer].[IntegrationAuthenticationId],  
                    [RevService].[Id] AS [RevServiceId],  
                    [RevService].[Number],  
                    [RevService].[ActivatedDate],  
                    CAST(ISNULL([RevServiceProduct].[CanceledDate], '9999-12-31T00:00:00') AS DATETIME) AS [DisconnectedDate],  
     [RevServiceProduct].[Status]  
                FROM [MobilityDevice]  
                INNER JOIN [RevService] ON [MobilityDevice].[MSISDN] = [RevService].[Number] AND [RevService].[IntegrationAuthenticationId] = @IntegrationAuthenticationId  
    LEFT JOIN [RevServiceProduct] ON [RevServiceProduct].[ServiceId] = [RevService].[RevServiceId]   
     AND [RevServiceProduct].[IsActive] = 1   
     AND [RevServiceProduct].[IsDeleted] = 0  
                    AND [RevServiceProduct].[IntegrationAuthenticationId] = [RevService].[IntegrationAuthenticationId]  
                INNER JOIN [dbo].[Integration_Authentication] ON [Integration_Authentication].[id] = [RevService].[IntegrationAuthenticationId]  
                LEFT JOIN [RevCustomer] ON [RevService].[RevCustomerId] = [RevCustomer].[id] AND [RevService].[IntegrationAuthenticationId] = [RevCustomer].[IntegrationAuthenticationId]  
                INNER JOIN [dbo].[Site] ON [Site].[RevCustomerId] = [RevCustomer].[Id]   
                    AND [Site].[TenantId] = [Integration_Authentication].[TenantId]  
                    AND [Site].[IsActive] = 1  
                    AND [Site].[IsDeleted] = 0  
                WHERE [MobilityDevice].[IsActive] = 1   
                    AND [MobilityDevice].[IsDeleted] = 0   
                    AND [Integration_Authentication].[IsActive] = 1   
                    AND [Integration_Authentication].[IsDeleted] = 0  
                    AND [RevCustomer].[IsActive] = 1  
                    AND [RevCustomer].[IsDeleted] = 0  
                    AND [RevService].[IsActive] = 1  
                    AND [RevService].[IsDeleted] = 0  
  
                UNION  
  
                SELECT [MobilityDevice].[id] AS [MobilityDeviceId],  
                    [Site].[TenantId],  
                    [Site].[id] AS [SiteId],  
                    [RevCustomer].[RevCustomerId] AS [AccountNumber],  
                    [RevCustomer].[IntegrationAuthenticationId],  
                    [RevService].[Id] AS [RevServiceId],  
                    [RevService].[Number],  
                    [RevService].[ActivatedDate],  
                    CAST(ISNULL([RevServiceProduct].[CanceledDate], '9999-12-31T00:00:00') AS DATETIME) AS [DisconnectedDate],  
     [RevServiceProduct].[Status]  
                FROM [MobilityDevice]  
                INNER JOIN [RevService] ON [MobilityDevice].[MSISDN] = '1' + [RevService].[Number] AND [RevService].[IntegrationAuthenticationId] = @IntegrationAuthenticationId  
    LEFT JOIN [RevServiceProduct] ON [RevServiceProduct].[ServiceId] = [RevService].[RevServiceId]   
     AND [RevServiceProduct].[IsActive] = 1   
     AND [RevServiceProduct].[IsDeleted] = 0  
                    AND [RevServiceProduct].[IntegrationAuthenticationId] = [RevService].[IntegrationAuthenticationId]  
                INNER JOIN [dbo].[Integration_Authentication] ON [Integration_Authentication].[id] = [RevService].[IntegrationAuthenticationId]  
                LEFT JOIN [RevCustomer] ON [RevService].[RevCustomerId] = [RevCustomer].[id] AND [RevService].[IntegrationAuthenticationId] = [RevCustomer].[IntegrationAuthenticationId]  
                INNER JOIN [dbo].[Site] ON [Site].[RevCustomerId] = [RevCustomer].[Id]   
                    AND [Site].[TenantId] = [Integration_Authentication].[TenantId]  
                    AND [Site].[IsActive] = 1  
                    AND [Site].[IsDeleted] = 0  
                WHERE [MobilityDevice].[IsActive] = 1   
                    AND [MobilityDevice].[IsDeleted] = 0  
                    AND [Integration_Authentication].[IsActive] = 1   
                    AND [Integration_Authentication].[IsDeleted] = 0  
                    AND [RevCustomer].[IsActive] = 1  
                    AND [RevCustomer].[IsDeleted] = 0  
                    AND [RevService].[IsActive] = 1  
                    AND [RevService].[IsDeleted] = 0  
            ) [DeviceTenant]  
            WHERE [Status] IS NOT NULL  
        ) [DeviceTenant1]  
        WHERE [RecordNumber] = 1  
    ) AS [SOURCE]  
    ON [TARGET].[TenantId] = [SOURCE].[TenantId] AND [TARGET].[MobilityDeviceId] = [SOURCE].[MobilityDeviceId]  
    WHEN NOT MATCHED THEN

            ,[TenantId]  
            ,[RevServiceId]  
            ,[AccountNumber]  
            ,[SiteId]  
            ,[AccountNumberIntegrationAuthenticationId]  
            ,[CreatedBy]  
            ,[CreatedDate]  
            ,[IsActive]  
            ,[IsDeleted]  
        )  
        VALUES

            [SOURCE].[MobilityDeviceId]  
            ,[SOURCE].[TenantId]  
            ,[SOURCE].[RevServiceId]  
            ,[SOURCE].[AccountNumber]  
            ,[SOURCE].[SiteId]  
            ,[SOURCE].[IntegrationAuthenticationId]  
            ,@processedBy  
            ,GETUTCDATE()  
            ,1  
            ,0  
        )  
    WHEN MATCHED AND ([TARGET].[SiteId] = [SOURCE].[SiteId] OR [SOURCE].[DisconnectedDate] IS NULL OR [SOURCE].[DisconnectedDate] = '9999-12-31T00:00:00') THEN  
        UPDATE  
        SET [TARGET].[RevServiceId] = [SOURCE].[RevServiceId]  
            ,[TARGET].[AccountNumber] = [SOURCE].[AccountNumber]  
            ,[TARGET].[SiteId] = [SOURCE].[SiteId]  
            ,[TARGET].[AccountNumberIntegrationAuthenticationId] = [SOURCE].[IntegrationAuthenticationId]  
            ,[TARGET].[ModifiedBy] = @processedBy  
            ,[TARGET].[ModifiedDate] = GETUTCDATE();  
  
    -- Clear Rev Service Association if Site does not match  
    UPDATE [Device_Tenant]  
    SET [RevServiceId] = NULL  
    FROM [Device_Tenant]  
    INNER JOIN [RevService] ON [RevService].[id] = [Device_Tenant].[RevServiceId]  
  AND [RevService].[IsActive] = 1   
  AND [RevService].[IsDeleted] = 0  
 INNER JOIN [Integration_Authentication] ON [Integration_Authentication].[id] = [IntegrationAuthenticationId]  
  AND [Integration_Authentication].[IsActive] = 1  
  AND [Integration_Authentication].[IsDeleted] = 0  
    INNER JOIN [Site] ON [RevService].[RevCustomerId] = [Site].[RevCustomerId]  
  AND [Site].[TenantId] = [Integration_Authentication].[TenantId]   
  AND [Site].[IsActive] = 1  
  AND [Site].[IsDeleted] = 0  
    WHERE [Site].[id] <> [Device_Tenant].[SiteId];  
  
    UPDATE [MobilityDevice_Tenant]  
    SET [RevServiceId] = NULL  
    FROM [MobilityDevice_Tenant]  
    INNER JOIN [RevService] ON [RevService].[id] = [MobilityDevice_Tenant].[RevServiceId]  
  AND [RevService].[IsActive] = 1   
  AND [RevService].[IsDeleted] = 0  
 INNER JOIN [Integration_Authentication] ON [Integration_Authentication].[id] = [IntegrationAuthenticationId]  
  AND [Integration_Authentication].[IsActive] = 1  
  AND [Integration_Authentication].[IsDeleted] = 0  
    INNER JOIN [Site] ON [RevService].[RevCustomerId] = [Site].[RevCustomerId]  
  AND [Site].[TenantId] = [Integration_Authentication].[TenantId]   
  AND [Site].[IsActive] = 1  
  AND [Site].[IsDeleted] = 0  
    WHERE [Site].[id] <> [MobilityDevice_Tenant].[SiteId];  
  
    -- Clear the Rev Service Association if the device has already been deleted  
    UPDATE [Device_Tenant]  
    SET [RevServiceId] = NULL  
    FROM [Device_Tenant]  
    INNER JOIN [Device] ON [Device].[id] = [Device_Tenant].[DeviceId]  
    WHERE [Device].[IsActive] = 0 AND [Device].[IsDeleted] = 1;  
  
    UPDATE [MobilityDevice_Tenant]  
    SET [RevServiceId] = NULL  
    FROM [MobilityDevice_Tenant]  
    INNER JOIN [MobilityDevice] ON [MobilityDevice].[id] = [MobilityDevice_Tenant].[MobilityDeviceId]  
    WHERE [MobilityDevice].[IsActive] = 0 AND [MobilityDevice].[IsDeleted] = 1;  
      
 -- Clear the Rev Service with existing MobilityDevice_Tenant when the telephone number changed in Rev IO  
 --For Mobility

   AND [RevService].[IsDeleted] = 0  
   AND [RevCustomer].[IsActive] = 1  
         AND [RevCustomer].[IsDeleted] = 0;  
   
  --For M2M

  INNER JOIN [RevService] ON [RevService].[id] = [Device_Tenant].[RevServiceId]  
   INNER JOIN [dbo].[Integration_Authentication] ON [Integration_Authentication].[id] = [RevService].[IntegrationAuthenticationId]  
  INNER JOIN [Device] ON [Device_Tenant].[DeviceId] = [Device].[id] AND [RevService].[IntegrationAuthenticationId] = @IntegrationAuthenticationId  
  INNER JOIN [RevCustomer] ON [RevService].[RevCustomerId] = [RevCustomer].[id] AND [RevService].[IntegrationAuthenticationId] = [RevCustomer].[IntegrationAuthenticationId]  
  WHERE (ISNULL([Device].[EID],'') <> [RevService].[Number] AND ISNULL([Device].[ICCID],'') <> [RevService].[Number])   
   AND ISNULL([Device].[MSISDN],'') <> [RevService].[Number]   
   AND ISNULL([Device].[MSISDN],'') <> '1' + [RevService].[Number]  
   AND [Device_Tenant].[IsActive] = 1 AND [Device_Tenant].[IsDeleted] = 0  
   AND [Device].[IsActive] = 1   
   AND [Device].[IsDeleted] = 0  
   AND [Integration_Authentication].[IsActive] = 1   
   AND [Integration_Authentication].[IsDeleted] = 0   
   AND [RevCustomer].[IsActive] = 1  
   AND [RevCustomer].[IsDeleted] = 0  
   AND [RevService].[IsActive] = 1  
   AND [RevService].[IsDeleted] = 0;  
  
    -- Set Default Site  
    INSERT INTO [Device_Tenant] ([DeviceId], [TenantId], [AccountNumber], [SiteId], [AccountNumberIntegrationAuthenticationId])  
    SELECT [Device].[id] AS [DeviceId], [Site].[TenantId], [RevCustomer].[RevCustomerId] AS [AccountNumber], [Site].[id] AS [SiteId], [RevCustomer].[IntegrationAuthenticationId]

    INNER JOIN [ServiceProvider] ON [Device].[ServiceProviderId] = [ServiceProvider].[Id]

    LEFT JOIN [RevCustomer] ON [agentSite].[RevCustomerId] = [RevCustomer].[id]  
    LEFT JOIN [Device_Tenant] [deviceTenant] ON [Device].[id] = [deviceTenant].[DeviceId] AND [agentSite].[TenantId] = [deviceTenant].[TenantId]

        AND [Device].[IsActive] = 1  
        AND [Device].[IsDeleted] = 0;

    INNER JOIN [ServiceProvider] ON [Device].[ServiceProviderId] = [ServiceProvider].[Id]  
    INNER JOIN [Site] ON [ServiceProvider].[TenantId] = [Site].[TenantId] AND [Site].[Id] = [Device_Tenant].[SiteId]

        AND [Device].[IsDeleted] = 0  
        AND [Device_Tenant].[SiteId] IS NULL;

    FROM [MobilityDevice]  
    INNER JOIN [ServiceProvider] ON [MobilityDevice].[ServiceProviderId] = [ServiceProvider].[Id]  
    INNER JOIN [Site] ON [ServiceProvider].[TenantId] = [Site].[TenantId] AND [Site].[IsDeleted] = 0 AND [Site].[IsSystemDefault] = 1  
    LEFT JOIN [RevCustomer] ON [Site].[RevCustomerId] = [RevCustomer].[id]  
    LEFT JOIN [MobilityDevice_Tenant] ON [MobilityDevice].[id] = [MobilityDevice_Tenant].[MobilityDeviceId] AND [Site].[TenantId] = [MobilityDevice_Tenant].[TenantId]

    INNER JOIN [ServiceProvider] ON [MobilityDevice].[ServiceProviderId] = [ServiceProvider].[Id]  
    INNER JOIN [Site] ON [ServiceProvider].[TenantId] = [Site].[TenantId] AND [Site].[IsDeleted] = 0 AND [Site].[IsSystemDefault] = 1 AND [Site].[TenantId] = [MobilityDevice_Tenant].[TenantId]  
    WHERE [MobilityDevice].[IsActive] = 1  
        AND [MobilityDevice].[IsDeleted] = 0  
        AND [MobilityDevice_Tenant].[SiteId] IS NULL;  
  
    INSERT INTO [MobilityDevice_Tenant] ([MobilityDeviceId], [TenantId], [AccountNumber], [SiteId], [AccountNumberIntegrationAuthenticationId])

    LEFT JOIN [RevCustomer] ON [agentSite].[RevCustomerId] = [RevCustomer].[id]  
    LEFT JOIN [MobilityDevice_Tenant] [deviceTenant] ON [MobilityDevice].[id] = [deviceTenant].[MobilityDeviceId] AND [agentSite].[TenantId] = [deviceTenant].[TenantId]  
    WHERE [deviceTenant].[id] IS NULL   
        AND [MobilityDevice].[IsActive] = 1  
        AND [MobilityDevice].[IsDeleted] = 0;

    INNER JOIN [Site] [agentSite] ON [agentSite].[ParentSiteId] = [Site].[Id] AND [agentSite].[IsSystemDefault] = 1 AND [agentSite].[TenantId] = [MobilityDevice_Tenant].[TenantId]  
    WHERE [MobilityDevice].[IsActive] = 1  
        AND [MobilityDevice].[IsDeleted] = 0  
        AND [MobilityDevice_Tenant].[SiteId] IS NULL;

  @IntegrationAuthenticationId INT  
AS

 SET NOCOUNT ON;

        RETURN;

      [PackageId],  
      [ServiceId],  
      [Description],  
      [Code1],  
      [Code2],

      [BilledThroughDate],  
      [CanceledDate],  
      [Status],  
      [StatusDate],  
      [StatusUserId],

      [Cost],  
      [WholesaleDescription],

      [FreeEndDate],  
      [Quantity],  
      [ContractStartDate],  
      [ContractEndDate],  
      [CreatedDate],  
      [TaxIncluded],  
      [GroupOnBill],  
      [Itemized],

          [RevServiceProductStaging].[ProductId],  
          [RevServiceProductStaging].[PackageId],

          [RevServiceProductStaging].[CanceledDate],  
          [RevServiceProductStaging].[Status],  
          [RevServiceProductStaging].[StatusDate],  
          [RevServiceProductStaging].[StatusUserId],  
          [RevServiceProductStaging].[ActivatedDate],  
          [RevServiceProductStaging].[Cost],  
          [RevServiceProductStaging].[WholesaleDescription],  
          [RevServiceProductStaging].[FreeStartDate],  
          [RevServiceProductStaging].[FreeEndDate],  
          [RevServiceProductStaging].[Quantity],  
          [RevServiceProductStaging].[ContractStartDate],  
          [RevServiceProductStaging].[ContractEndDate],  
          [RevServiceProductStaging].[CreatedDate],  
          [RevServiceProductStaging].[TaxIncluded],  
          [RevServiceProductStaging].[GroupOnBill],  
          [RevServiceProductStaging].[Itemized],  
                [RevServiceProductStaging].[IntegrationAuthenticationId],  
                ROW_NUMBER() OVER(PARTITION BY [ServiceProductId] ORDER BY [RevServiceProductStaging].[RevServiceProductStagingId] DESC) AS [RecordNumber]  
            FROM [dbo].[RevServiceProductStaging]  
         ) [RevServiceProductStg]  
         WHERE [RecordNumber] = 1  
    ) AS [SOURCE]  
    ON [SOURCE].[ServiceProductId] = [TARGET].[ServiceProductId] AND [SOURCE].[IntegrationAuthenticationId] = [TARGET].[IntegrationAuthenticationId]  
    WHEN NOT MATCHED BY TARGET THEN  
        INSERT (  
   [ServiceProductId],  
   [CustomerId],  
   [ProductId],  
   [PackageId],  
   [ServiceId],  
   [Description],  
   [Code1],  
   [Code2],  
   [Rate],  
   [BilledThroughDate],  
   [CanceledDate],  
   [Status],  
   [StatusDate],  
   [StatusUserId],  
   [ActivatedDate],  
   [Cost],  
   [WholesaleDescription],  
   [FreeStartDate],  
   [FreeEndDate],  
   [Quantity],  
   [ContractStartDate],  
   [ContractEndDate],  
   [CreatedDate],  
   [TaxIncluded],  
   [GroupOnBill],  
   [Itemized],  
   [CreatedBy],  
   [ModifiedBy],  
   [ModifiedDate],  
   [DeletedBy],  
   [DeletedDate],  
   [IsDeleted],  
   [IsActive],  
   [IntegrationAuthenticationId]  
  )

            [SOURCE].[ServiceProductId],  
      [SOURCE].[CustomerId],  
      [SOURCE].[ProductId],  
      [SOURCE].[PackageId],  
      [SOURCE].[ServiceId],  
      [SOURCE].[Description],  
      [SOURCE].[Code1],  
      [SOURCE].[Code2],  
      [SOURCE].[Rate],  
      [SOURCE].[BilledThroughDate],  
      [SOURCE].[CanceledDate],  
      [SOURCE].[Status],  
      [SOURCE].[StatusDate],  
      [SOURCE].[StatusUserId],  
      [SOURCE].[ActivatedDate],  
      [SOURCE].[Cost],  
      [SOURCE].[WholesaleDescription],  
      [SOURCE].[FreeStartDate],  
      [SOURCE].[FreeEndDate],  
      [SOURCE].[Quantity],  
      [SOURCE].[ContractStartDate],  
      [SOURCE].[ContractEndDate],  
      [SOURCE].[CreatedDate],  
      [SOURCE].[TaxIncluded],  
      [SOURCE].[GroupOnBill],  
      [SOURCE].[Itemized],  
      @processedBy,  
      NULL,  
      NULL,  
      NULL,  
      NULL,  
      0,  
      1,  
      [SOURCE].[IntegrationAuthenticationId]  
        )  
    WHEN NOT MATCHED BY SOURCE AND [TARGET].[IntegrationAuthenticationId] = @IntegrationAuthenticationId THEN  
        UPDATE  
     SET  
         [TARGET].[IsDeleted] = 1,  
         [TARGET].[IsActive] = 0,  
         [TARGET].[ModifiedBy] = @processedBy,  
         [TARGET].[ModifiedDate] = GETUTCDATE()  
    WHEN MATCHED AND (  
                        TARGET.[CustomerId] <> SOURCE.[CustomerId]  
                        OR TARGET.[ProductId] <> SOURCE.[ProductId]   
                        OR TARGET.[PackageId] <> SOURCE.[PackageId]   
                        OR TARGET.[ServiceId] <> SOURCE.[ServiceId]  
                        OR TARGET.[Description] <> SOURCE.[Description]  
                        OR TARGET.[Rate] <> SOURCE.[Rate]  
      OR TARGET.[CanceledDate] <> SOURCE.[CanceledDate]  
      OR TARGET.[Status] <> SOURCE.[Status]  
      OR TARGET.[StatusDate] <> SOURCE.[StatusDate]  
      OR TARGET.[BilledThroughDate] <> SOURCE.[BilledThroughDate]  
      OR TARGET.[StatusUserId] <> SOURCE.[StatusUserId]  
      OR TARGET.[ActivatedDate] <> SOURCE.[ActivatedDate]  
      OR TARGET.[Cost] <> SOURCE.[Cost]  
      OR TARGET.[Quantity] <> SOURCE.[Quantity]  
      OR TARGET.[ContractStartDate] <> SOURCE.[ContractStartDate]  
      OR TARGET.[ContractEndDate] <> SOURCE.[ContractEndDate]  
      OR TARGET.[ActivatedDate] <> SOURCE.[ActivatedDate]  
      OR (TARGET.IsDeleted = 1)  
      OR (TARGET.IsActive = 0)) THEN  
        UPDATE  
     SET  
      [TARGET].[CustomerId] = [SOURCE].[CustomerId],  
      [TARGET].[ProductId] = [SOURCE].[ProductId],  
      [TARGET].[PackageId] = [SOURCE].[PackageId],  
      [TARGET].[ServiceId] = [SOURCE].[ServiceId],  
      [TARGET].[Description]= [SOURCE].[Description],  
      [TARGET].[Code1] = [SOURCE].[Code1],  
      [TARGET].[Code2]= [SOURCE].[Code2],

            [RevPackageId]  
        )  
        VALUES(  
            [SOURCE].[PackageProductId]  
            ,[SOURCE].[ProductId]  
            ,[SOURCE].[PackageId]  
            ,[SOURCE].[Description]  
            ,[SOURCE].[Code1]  
            ,[SOURCE].[Code2]  
            ,[SOURCE].[Rate]  
           ,[SOURCE].[Cost]  
            ,[SOURCE].[BuyRate]  
            ,[SOURCE].[Quantity]  
            ,[SOURCE].[TaxIncluded]  
            ,[SOURCE].[GroupOnBill]  
            ,[SOURCE].[Itemized]  
            ,[SOURCE].[Credit]  
            ,@processedBy  
            ,GETUTCDATE()  
            ,NULL  
            ,NULL  
            ,NULL  
            ,NULL  
            ,0  
            ,1  
            ,[SOURCE].[IntegrationAuthenticationId]  
            ,[SOURCE].[RevProductId]  
            ,[SOURCE].[RevPackageId]  
        )  
    WHEN NOT MATCHED BY SOURCE AND [TARGET].[IntegrationAuthenticationId] = @IntegrationAuthenticationId THEN  
        UPDATE  
     SET  
         [TARGET].[IsDeleted] = 1,  
         [TARGET].[IsActive] = 0,  
         [TARGET].[ModifiedBy] = @processedBy,  
         [TARGET].[ModifiedDate] = GETUTCDATE()  
    WHEN MATCHED THEN  
        UPDATE  
        SET [TARGET].[ProductId] = [SOURCE].[ProductId],  
            [TARGET].[PackageId] = [SOURCE].[PackageId],  
            [TARGET].[Description] = [SOURCE].[Description],  
            [TARGET].[Code1] = [SOURCE].[Code1],  
            [TARGET].[Code2] = [SOURCE].[Code2],  
            [TARGET].[Rate] = [SOURCE].[Rate],  
            [TARGET].[Cost] = [SOURCE].[Cost],  
            [TARGET].[BuyRate] = [SOURCE].[BuyRate],  
            [TARGET].[Quantity] = [SOURCE].[Quantity],  
            [TARGET].[TaxIncluded] = [SOURCE].[TaxIncluded],  
            [TARGET].[GroupOnBill] = [SOURCE].[GroupOnBill],  
            [TARGET].[Itemized] = [SOURCE].[Itemized],  
            [TARGET].[Credit] = [SOURCE].[Credit],  
            [TARGET].[ModifiedBy] = @processedBy,  
            [TARGET].[ModifiedDate] = GETUTCDATE(),  
            [TARGET].[IsDeleted] = 0,  
         [TARGET].[IsActive] = 1,  
            [TARGET].[IntegrationAuthenticationId] = [SOURCE].[IntegrationAuthenticationId],  
            [TARGET].[RevProductId] = [SOURCE].[RevProductId],  
            [TARGET].[RevPackageId] = [SOURCE].[RevPackageId];  
    END;


usp_Rev_Get_Next_Authentication

CREATE PROCEDURE [dbo].[usp_Rev_Get_Next_Authentication]  
 @integrationAuthenticationId INT  
AS  
BEGIN  
 SET NOCOUNT ON;  
   
 SELECT  
  ISNULL([NextIntegrationAuthenticationId], -1) AS [NextIntegrationAuthenticationId]  
 FROM  
 (   
  SELECT  
   TOP (1)   
   [IntegrationAuthentication].[id] AS [NextIntegrationAuthenticationId]  
  FROM  
   [dbo].[Integration_Authentication] [IntegrationAuthentication]  
  WHERE  
   [IntegrationAuthentication].[IntegrationId] = 3  
   AND  
   [TokenValue] IS NOT NULL  
   AND  
   [IntegrationAuthentication].[IsActive] = 1  
   AND  
   [IntegrationAuthentication].[IsDeleted] = 0  
   AND  
   [IntegrationAuthentication].[id] > @integrationAuthenticationId  
  ORDER BY   
   [IntegrationAuthentication].[id]  
 ) [NextIntegrationAuthentication];  
END
